# 🚀 取名系统插件化重构详细计划

## 📋 重构概述

本计划旨在将当前的硬编码耦合架构重构为配置驱动的插件化架构，支持模块的动态加载、配置和管理，提升系统的可扩展性和维护性。

## 🎯 重构目标

### 核心目标
- ✅ 实现真正的插件化架构
- ✅ 支持模块的热插拔
- ✅ 配置驱动的模块管理
- ✅ 统一的数据模型和接口
- ✅ 支持确定性等级的动态切换

### 技术目标
- 🔧 零循环依赖
- 🔧 模块间低耦合
- 🔧 接口标准化
- 🔧 配置外部化
- 🔧 类型安全保证

```typescript
// 1.1 定义核心插件接口
interface NamingPlugin {
  readonly id: string;
  readonly version: string;
  readonly layer: 1 | 2 | 3 | 4;
  readonly dependencies: PluginDependency[];
  readonly metadata: PluginMetadata;
  
  initialize(config: PluginConfig, context: PluginContext): Promise<void>;
  process(input: PluginInput): Promise<PluginOutput>;
  validate(input: PluginInput): ValidationResult;
  destroy(): Promise<void>;
}

// 1.2 设计依赖注入容器
class PluginContainer {
  private plugins: Map<string, NamingPlugin>;
  private dependencyGraph: DependencyGraph;
  private configManager: ConfigManager;
  
  register(plugin: NamingPlugin): Promise<void>;
  unregister(pluginId: string): Promise<void>;
  resolve<T extends NamingPlugin>(pluginId: string): T;
  getExecutionOrder(level: CertaintyLevel): string[];
}

// 1.3 统一数据模型
interface StandardInput {
  readonly requestId: string;
  readonly certaintyLevel: CertaintyLevel;
  readonly data: InputData;
  readonly context: ProcessingContext;
}

interface StandardOutput {
  readonly pluginId: string;
  readonly results: OutputData;
  readonly confidence: number;
  readonly metadata: OutputMetadata;
  readonly errors?: PluginError[];
}
```

```typescript
// 6.1 处理管道
class NamingPipeline {
  constructor(
    private container: PluginContainer,
    private configManager: ConfigManager
  ) {}
  
  async execute(
    input: PipelineInput,
    level: CertaintyLevel
  ): Promise<PipelineOutput> {
    const context = new ProcessingContext(input.requestId);
    const enabledPlugins = this.configManager.getEnabledPlugins(level);
    const executionOrder = this.container.getExecutionOrder(enabledPlugins);
    
    const results = new Map<string, PluginOutput>();
    const errors: PluginError[] = [];
    
    // 按层级顺序执行
    for (const layer of [1, 2, 3, 4]) {
      const layerPlugins = executionOrder.filter(id => 
        this.container.getPlugin(id).layer === layer
      );
      
      // 检查是否可以并行执行
      const canParallel = this.canExecuteInParallel(layerPlugins, level);
      
      if (canParallel) {
        await this.executeParallel(layerPlugins, input, context, results, errors);
      } else {
        await this.executeSequential(layerPlugins, input, context, results, errors);
      }
    }
    
    return this.buildPipelineOutput(results, errors, context);
  }
  
  private async executeWithFallback(
    pluginId: string,
    input: StandardInput,
    context: ProcessingContext
  ): Promise<PluginOutput> {
    const plugin = this.container.getPlugin(pluginId);
    
    try {
      // 验证依赖
      const dependencyCheck = await this.validateDependencies(plugin, context);
      if (!dependencyCheck.valid) {
        return this.handleMissingDependencies(plugin, dependencyCheck);
      }
      
      // 执行插件
      const output = await plugin.process(input);
      
      // 验证输出
      const validationResult = await this.validateOutput(output);
      if (!validationResult.valid) {
        throw new PluginError('INVALID_OUTPUT', validationResult.errors.join(', '));
      }
      
      return output;
      
    } catch (error) {
      // 尝试降级处理
      const fallbackResult = await this.tryFallback(plugin, input, context, error);
      if (fallbackResult) {
        return fallbackResult;
      }
      
      throw error;
    }
  }
  
  private async tryFallback(
    plugin: NamingPlugin,
    input: StandardInput,
    context: ProcessingContext,
    error: any
  ): Promise<PluginOutput | null> {
    const fallbackConfig = this.configManager.getFallbackConfig(plugin.id);
    
    if (fallbackConfig?.fallbackPlugin) {
      console.warn(`插件 ${plugin.id} 失败，尝试降级到 ${fallbackConfig.fallbackPlugin}`);
      const fallbackPlugin = this.container.getPlugin(fallbackConfig.fallbackPlugin);
      return await fallbackPlugin.process(input);
    }
    
    if (fallbackConfig?.defaultValue) {
      console.warn(`插件 ${plugin.id} 失败，使用默认值`);
      return this.createDefaultOutput(plugin.id, fallbackConfig.defaultValue);
    }
    
    return null;
  }
}

class PluginContainer {
  private plugins: Map<string, NamingPlugin> = new Map();
  private dependencyGraph: DependencyGraph = new DependencyGraph();
  private lifecycleManager: PluginLifecycleManager = new PluginLifecycleManager();
  
  async register(plugin: NamingPlugin): Promise<void> {
    // 验证插件
    const validation = await this.validatePlugin(plugin);
    if (!validation.valid) {
      throw new PluginError('INVALID_PLUGIN', validation.errors.join(', '));
    }
    
    // 检查依赖
    const dependencyCheck = this.checkDependencies(plugin);
    if (!dependencyCheck.satisfied) {
      throw new PluginError('MISSING_DEPENDENCIES', 
        `Missing: ${dependencyCheck.missing.join(', ')}`
      );
    }
    
    // 注册到依赖图
    this.dependencyGraph.addNode(plugin.id, plugin.dependencies);
    
    // 检查循环依赖
    if (this.dependencyGraph.hasCycles()) {
      this.dependencyGraph.removeNode(plugin.id);
      throw new PluginError('CIRCULAR_DEPENDENCY', 
        `Circular dependency detected involving ${plugin.id}`
      );
    }
    
    // 初始化插件
    await plugin.initialize(
      this.configManager.getPluginConfig(plugin.id),
      this.createPluginContext(plugin.id)
    );
    
    // 注册成功
    this.plugins.set(plugin.id, plugin);
    this.lifecycleManager.markAsActive(plugin.id);
    
    console.log(`✅ 插件 ${plugin.id} 注册成功`);
  }
  
  async unregister(pluginId: string): Promise<void> {
    const plugin = this.plugins.get(pluginId);
    if (!plugin) return;
    
    // 检查是否有其他插件依赖此插件
    const dependents = this.dependencyGraph.getDependents(pluginId);
    if (dependents.length > 0) {
      throw new PluginError('PLUGIN_IN_USE', 
        `Plugin ${pluginId} is required by: ${dependents.join(', ')}`
      );
    }
    
    // 清理插件
    await plugin.destroy();
    this.plugins.delete(pluginId);
    this.dependencyGraph.removeNode(pluginId);
    this.lifecycleManager.markAsInactive(pluginId);
    
    console.log(`🗑️ 插件 ${pluginId} 卸载完成`);
  }
  
  getExecutionOrder(enabledPluginIds: string[]): string[] {
    return this.dependencyGraph.topologicalSort(enabledPluginIds);
  }
}
```
