/**
 * 增强型三才五格计算器
 * 基于qiming项目的算法，添加详细解释和可视化支持
 */
export class EnhancedSancaiCalculator {
  private dataLoader: EnhancedDataLoader;
  private sancaiRules: SancaiRule[];
  
  constructor() {
    this.dataLoader = EnhancedDataLoader.getInstance();
  }
  
  // 完整的三才五格计算
  async calculateComplete(
    familyName: string, 
    firstName: string, 
    lastName: string,
    options: CalculationOptions = {}
  ): Promise<CompleteSancaiResult> {
    
    // 1. 获取笔画数
    const strokes = await this.getCharacterStrokes(
      [familyName, firstName, lastName], 
      options.useTraditional
    );
    
    // 2. 计算五格
    const grids = this.calculateGrids(strokes);
    
    // 3. 计算三才
    const sancai = this.calculateSancai(grids);
    
    // 4. 生成详细解释
    const explanation = this.generateExplanation(grids, sancai);
    
    // 5. 计算综合评分
    const score = this.calculateOverallScore(grids, sancai);
    
    return {
      grids,
      sancai, 
      explanation,
      score,
      visualizationData: this.generateVisualizationData(grids, sancai)
    };
  }
  
  // 生成可视化数据
  private generateVisualizationData(grids: GridResult, sancai: SancaiResult): VisualizationData {
    return {
      radarChartData: this.createRadarData(grids),
      progressRingData: this.createProgressData(sancai),
      flowChartData: this.createFlowData(grids, sancai)
    };
  }
}

/**
 * 增强型名字生成API
 */
export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse<EnhancedGenerationResponse>
) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }
  
  try {
    const { familyName, gender, preferences, options } = req.body;
    
    // 参数验证
    const validation = validateGenerationRequest(req.body);
    if (!validation.isValid) {
      return res.status(400).json({ error: validation.errors });
    }
    
    // 初始化生成器
    const generator = new EnhancedNameGenerator();
    await generator.initialize();
    
    // 生成名字
    const startTime = Date.now();
    const results = await generator.generateNames({
      familyName,
      gender,
      preferences,
      options
    });
    const generationTime = Date.now() - startTime;
    
    // 生成统计信息
    const statistics = generateStatistics(results, generationTime);
    
    // 生成详细解释
    const explanations = await generateExplanations(results);
    
    res.status(200).json({
      success: true,
      data: {
        names: results,
        statistics,
        explanations,
        metadata: {
          requestId: generateRequestId(),
          timestamp: new Date().toISOString(),
          version: '2.0'
        }
      }
    });
    
  } catch (error) {
    console.error('Name generation error:', error);
    res.status(500).json({ 
      error: 'Internal server error',
      message: process.env.NODE_ENV === 'development' ? error.message : undefined
    });
  }
}
