## 项目概况对比

### child_name项目（主项目）
- **技术栈**: Next.js + TypeScript + Tailwind CSS
- **架构**: 前端全栈应用，静态数据文件
- **特色**: 诗词取名、文化内涵、现代化UI
- **算法**: 基础的三才五格、五行评分、音韵美学

- **技术栈**: Go语言 + xorm + chronos
- **架构**: 命令行工具 + 数据库驱动
- **特色**: 专业八字分析、生肖用字、多重过滤
- **算法**: 完整的传统姓名学算法体系

// 评分算法
func (z *Zodiac) Point(char string) int {
    score := 3 // 基础分
    if strings.Contains(z.Ji, char) {
        score -= 3 // 忌用字扣3分
    }
    if strings.Contains(z.Xi, char) {
        score += 2 // 喜用字加2分
    }
    return score
}

// 生肖评分器
class ZodiacScorer {
  private zodiacData: Map<string, ZodiacCharData>;

  getZodiacScore(char: string, birthYear: number): number {
    const zodiac = this.getZodiacByYear(birthYear);
    const data = this.zodiacData.get(zodiac);
    
    let score = 3; // 基础分
    if (data.jiChars.has(char)) score -= 3;
    if (data.xiChars.has(char)) score += 2;
    
    return Math.max(0, score); // 最低0分
  }

  filterByZodiac(names: GeneratedName[], birthYear: number): GeneratedName[] {
    return names.filter(name => {
      const totalScore = name.chars.reduce((sum, char) => 
        sum + this.getZodiacScore(char, birthYear), 0);
      return totalScore >= name.chars.length * 3; // 平均分不低于3
    });
  }
}

class ConcurrentNameGenerator {
  async* generateNamesAsync(config: NameConfig): AsyncGenerator<GeneratedName> {
    // 异步生成候选名字
    const strokeCombinations = await this.getStrokeCombinations(config);
    
    for (const combo of strokeCombinations) {
      const chars = await this.getCharsByStrokes(combo);
      
      // 使用Web Worker进行并发处理（如果需要）
      yield* this.generateCombinations(chars, config);
    }
  }

  private async* generateCombinations(
    chars: CharacterData[], 
    config: NameConfig
  ): AsyncGenerator<GeneratedName> {
    for (const char1 of chars.first) {
      for (const char2 of chars.second) {
        const name = this.createName(char1, char2, config);
        if (await this.passesBasicFilters(name)) {
          yield name;
        }
      }
    }
  }
}

#### 主项目质量保证系统
```typescript
// 质量等级定义
type QualityGrade = 'excellent' | 'premium' | 'good' | 'average' | 'basic';

interface QualityAssessment {
  grade: QualityGrade;
  score: number;        // 0-100综合评分
  factors: {
    wuge: number;       // 五格评分
    sancai: number;     // 三才评分
    bazi: number;       // 八字匹配度
    zodiac: number;     // 生肖适配度
    cultural: number;   // 文化内涵
    phonetic: number;   // 音韵美感
  };
  strengths: string[];  // 优势说明
  concerns: string[];   // 潜在问题
  recommendations: string[]; // 改进建议
}

class QualityController {
  // 硬过滤：确保基本质量
  hardFilter(name: GeneratedName, config: EnhancedNamingConfig): boolean {
    if (config.wuge.hardFilter) {
      const wuge = this.calculateWuge(name);
      return this.allGeJi(wuge); // 所有格都必须为吉
    }
    
    if (config.bazi.enabled && config.bazi.requireXiYong) {
      return this.containsXiYong(name, config.birthTime);
    }
    
    if (config.zodiac.enabled && config.zodiac.excludeJi) {
      return !this.containsJiChar(name, config.birthYear);
    }
    
    return true;
  }

  // 软过滤：质量评分
  assessQuality(name: GeneratedName, config: EnhancedNamingConfig): QualityAssessment {
    const factors = {
      wuge: this.calculateWugeScore(name),
      sancai: this.calculateSancaiScore(name),
      bazi: config.bazi.enabled ? this.calculateBaziScore(name, config.birthTime) : 80,
      zodiac: config.zodiac.enabled ? this.calculateZodiacScore(name, config.birthYear) : 80,
      cultural: this.calculateCulturalScore(name),
      phonetic: this.calculatePhoneticScore(name)
    };

    const score = this.calculateOverallScore(factors, config);
    const grade = this.determineGrade(score);

    return {
      grade,
      score,
      factors,
      strengths: this.identifyStrengths(factors),
      concerns: this.identifyConcerns(factors),
      recommendations: this.generateRecommendations(factors, config)
    };
  }

  // 分级处理
  categorizeByQuality(names: GeneratedName[]): Map<QualityGrade, GeneratedName[]> {
    const categories = new Map();
    
    for (const name of names) {
      const assessment = this.assessQuality(name, this.config);
      const grade = assessment.grade;
      
      if (!categories.has(grade)) {
        categories.set(grade, []);
      }
      categories.get(grade).push({
        ...name,
        quality: assessment
      });
    }
    
    return categories;
  }
}
```
